*) 矩阵
  mat4 gl_ModelViewMatrix;
  mat4 gl_ProjectionMatrix;
  mat4 gl_ModelViewProjectionMatrix;
  mat3 gl_NormalMatrix;
  mat4 gl_TextureMatrix[gl_MaxTextureCoords];
  
*) 普通缩放
  float gl_NormalScale;
 
*) 窗口坐标中的深度范围
  struct gl_DepthRangleParameters
  {
    float near;
    float far;
    float diff;  /* far-near */
  };
  
  struct gl_DepthRangleParameters gl_DepthRangle; 
  
*) 载剪平面
  vec4 gl_ClipPlane[gl_MaxClipPlanes];
 

*) 点大小
  struct gl_PointParameters
  {
    float size;
    float sizeMin;
    float sizeMax;
    float fadeThresholdSize;
    float distanceConstantAttenuation;
    float distanceLinearAttenuation;
    float distanceQuadraticAttenuation;
  };
  
*) 材质状态
  struct gl_MaterialParameters
  {
    vec4 emission;
    vec4 ambient;
    vec4 diffuse;
    vec4 specular;
    float shininess;
  };
  
  struct gl_MaterialParameters gl_FrontMaterial;
  struct gl_MaterialParametersS gl_BackMaterial;

*) 光状态
  struct gl_LightSourceParameters 
  {
    vec4 ambient;
    vec4 diffuse;
    vec4 specular;
    vec4 position;
    vec4 halfVector;
    vec3 spotDirection;
    float spotExponent;
    float spotCutoff;
    float spotCosCutoff;
    float constantAttenuation;
    float linearAttenuation;
    float quadraticAttenuation;
  };
  
  struct gl_LightSourceParameters gl_LightSource[gl_MaxLights];
  struct gl_LightModelParameters
  {
    vec4 ambient;
  };
  struct gl_LightModelParameters gl_LightModel;
  
*) 材质
  struct gl_LightModelProducts
  {
    vec4 sceneColor;
  };
  struct gl_LightModelProducts gl_FrontLightModelProduct;
  struct gl_lightModelProducts gl_BackLightModelProduct;
  struct gl_LightProducts
  {
    vec4 ambient;
    vec4 diffuse;
    vec4 specular;
  };
  struct gl_LightProducts gl_FrontLightProduct[gl_MaxLights];
  struct gl_LightProducts gl_BackLightProduct[gl_MaxLights];
  
*) 纹理环境和生成
  vec4 gl_TextureEnvColor[gl_MaxTextureImageUnits];
  vec4 gl_EyePlaneS[gl_MaxTextureCoords];
  vec4 gl_EyePlaneT[gl_MaxTextureCoords];
  vec4 gl_EyePlaneR[gl_MaxTextureCoords];
  vec4 gl_EyePlaneQ[gl_MaxTextureCoords];
  vec4 gl_ObjectPlaneS[gl_MaxTextureCoords];
  vec4 gl_ObjectPlaneT[gl_MaxTextureCoords];
  vec4 gl_ObjectPlaneR[gl_MaxTextureCoords];
  vec4 gl_ObjectPlaneQ[gl_MaxTextureCoords];
 
 
 *) 雾化
  struct gl_FogParameters
  {
    vec4 color;
    float density;
    float start;
    float end;
    float scale; /* 1/(end-start) */
  }
  struct gl_FogParameters gl_Fog;
